Table cabinet_diabetestype [note: 'Справочник типов диабета'] {
  id integer [primary key]
  name varchar(100) [unique, not null]
  created_at timestamp [not null]
  updated_at timestamp [not null]
}

Table cabinet_treatmenttype [note: 'Справочник типов лечения'] {
  id integer [primary key]
  name varchar(100) [unique, not null]
  created_at timestamp [not null]
  updated_at timestamp [not null]
}

Table cabinet_disease [note: 'Справочник сопутствующих заболеваний'] {
  id integer [primary key]
  name varchar(100) [unique, not null]
  created_at timestamp [not null]
  updated_at timestamp [not null]
}

Table cabinet_diseaseappuser [note: 'Связка сопутствующего заболевания и пользователя'] {
  id integer [primary key]
  user_id integer [not null, ref: > auth_service_appuser.id]
  disease_id integer [not null, ref: > cabinet_disease.id]
  created_at timestamp [not null]
  updated_at timestamp [not null]
  indexes {
    user_id
    disease_id
  }
}

Table auth_service_town [note: 'Город'] {
  id integer [primary key]
  name varchar(100) [unique, not null]
  timestamp integer [not null]
  created_at timestamp [not null]
  updated_at timestamp [not null]
}

Table auth_service_appuser [note: 'Пользователь'] {
  id integer [primary key]
  username varchar(150) [not null, unique]
  first_name varchar(150) [not null, default: '']
  last_name varchar(150) [not null, default: '']
  email varchar [not null, default: '']
  phone_number varchar(15) [not null, default: '']
  is_staff boolean [not null, default: false]
  is_active boolean [not null, default: true]
  town_id integer [not null, ref: > auth_service_town.id]
  date_joined timestamp [not null]
  image varchar [null]
  patronymic_name varchar(150) [not null, default: '']
  gender varchar(10) [not null, default: 'undefined']
  height integer [null]
  birth_date date [null]
  diabetes_type_id integer [ref: > cabinet_diabetestype.id, null]
  treatment_type_id integer [ref: > cabinet_treatmenttype.id, null]
  created_at timestamp [not null]
  updated_at timestamp [not null]
  indexes {
    diabetes_type_id
    treatment_type_id
  }
}
